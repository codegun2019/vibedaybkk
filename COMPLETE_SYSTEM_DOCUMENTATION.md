BEGIN
    UPDATE models m
    SET 
        m.booking_count = (SELECT COUNT(*) FROM bookings WHERE model_id = p_model_id AND status = 'completed'),
        m.rating = (SELECT COALESCE(AVG(rating), 0) FROM bookings WHERE model_id = p_model_id AND rating IS NOT NULL)

END
